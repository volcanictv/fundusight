#   pip install torch==2.11.0+cu128 torchvision==0.26.0+cu128 --index-url https://download.pytorch.org/whl/cu128
# then run `pip install -r requirements.txt` — pip matches the public version
# (2.11.0/0.26.0) regardless of the +cu128 local build tag, so it's satisfied
# and leaves the CUDA build in place.

# Streamlit Cloud's packages.txt installer doesn't strip '#' lines and
# instead tries to apt-get install every word in them literally.
